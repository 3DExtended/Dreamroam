62  void BeginSession(const std::string& name,
63  const std::string& filepath = "profilingResults.json") {
64  std::lock_guard lock(m_Mutex);
65  if (m_CurrentSession) {
66  // If there is already a current session, then close it before
67  // beginning new one. Subsequent profiling output meant for the
68  // original session will end up in the newly opened session instead.
69  // That's better than having badly formatted profiling output.
70  InternalEndSession();
71  }
72  m_OutputStream.open(filepath);
73 
74  if (m_OutputStream.is_open()) {
75  m_CurrentSession = new InstrumentationSession({name});
76  WriteHeader();
77  } else {
78  DR_ASSERT(false, "Could not open file stream.");
79  }
80  }
81 
82  void EndSession() {
83  std::lock_guard lock(m_Mutex);
84  InternalEndSession();
85  }
86 
87  void WriteProfile(const ProfileResult& result) {
88  std::stringstream json;
89 
90  json << std::setprecision(3) << std::fixed;
91  json << ",{";
92  json << "\"cat\":\"function\",";
93  json << "\"dur\":" << (result.ElapsedTime.count()) << ',';
94  json << "\"name\":\"" << result.Name << "\",";
95  json << "\"ph\":\"X\",";
96  json << "\"pid\":0,";
97  json << "\"tid\":" << result.ThreadID << ",";
98  json << "\"ts\":" << result.Start.count();
99  json << "}";
100 
101  std::lock_guard lock(m_Mutex);
102  if (m_CurrentSession) {
103  m_OutputStream << json.str();
104  m_OutputStream.flush();
105  }
106  }
107 
108  static Instrumentor& Get() {
109  static Instrumentor instance;
110  return instance;
111  }
112 
113 private:
114  Instrumentor() : m_CurrentSession(nullptr) {}
115 
116  ~Instrumentor() { EndSession(); }
117 
118  void WriteHeader() {
119  m_OutputStream << "{\"otherData\": {},\"traceEvents\":[{}";
120  m_OutputStream.flush();
121  }
122 
123  void WriteFooter() {
124  m_OutputStream << "]}";
125  m_OutputStream.flush();
126  }
127 
128  // Note: you must already own lock on m_Mutex before
129  // calling InternalEndSession()
130  void InternalEndSession() {
131  if (m_CurrentSession) {
132  WriteFooter();
133  m_OutputStream.close();
134  delete m_CurrentSession;
135  m_CurrentSession = nullptr;
136  }
137  }
138 
139 private:

141  InstrumentationSession* m_CurrentSession;
142  std::ofstream m_OutputStream;

145 class InstrumentationTimer {
146 public:
147  InstrumentationTimer(const char* name) : m_Name(name), m_Stopped(false) {
148  m_StartTimepoint = std::chrono::steady_clock::now();
149  }
150 
151  ~InstrumentationTimer() {
152  if (!m_Stopped) Stop();
153  }
154 
155  void Stop() {
156  auto endTimepoint = std::chrono::steady_clock::now();
157  auto highResStart =
158  FloatingPointMicroseconds{m_StartTimepoint.time_since_epoch()};
159  auto elapsedTime =
160  std::chrono::time_point_cast<std::chrono::microseconds>(
161  endTimepoint)
162  .time_since_epoch() -
163  std::chrono::time_point_cast<std::chrono::microseconds>(
164  m_StartTimepoint)
165  .time_since_epoch();
166 
167  Instrumentor::Get().WriteProfile(
168  {m_Name, highResStart, elapsedTime, std::this_thread::get_id()});
169 
170  m_Stopped = true;
171  }
172 
173 private:
174  const char* m_Name;
175  std::chrono::time_point<std::chrono::steady_clock> m_StartTimepoint;
176  bool m_Stopped;
177 };

185 template <size_t N, size_t K>
186 constexpr auto CleanupOutputString(const char (&expr)[N],
187  const char (&remove)[K]) {
188  ChangeResult<N> result = {};
189 
190  size_t srcIndex = 0;
191  size_t dstIndex = 0;
192  while (srcIndex < N) {
193  size_t matchIndex = 0;
194  while (matchIndex < K - 1 && srcIndex + matchIndex < N - 1 &&
195  expr[srcIndex + matchIndex] == remove[matchIndex])
196  matchIndex++;
197  if (matchIndex == K - 1) srcIndex += matchIndex;
198  result.Data[dstIndex++] = expr[srcIndex] == '"' ? '\'' : expr[srcIndex];
199  srcIndex++;
200  }
201  return result;
202 }
